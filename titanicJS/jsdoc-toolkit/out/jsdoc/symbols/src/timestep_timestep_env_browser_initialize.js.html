1 "use import";
  2 
  3 import ...device;
  4 from util.browser import $;
  5 
  6 var onResize;
  7 
  8 if (device.isMobileBrowser) {
  9 	var screen = device.screen;
 10 	if (device.isIOS || device.isAndroid) {
 11 		
 12 		var _isFocused = false;
 13 		window.addEventListener('focus', function(e) {
 14 			var tag = e.target.tagName;
 15 			if (tag == 'TEXTAREA' || tag == 'INPUT') {
 16 				_isFocused = true;
 17 			}
 18 		}, true);
 19 		
 20 		document.addEventListener('blur', function(e) {
 21 			if (_isFocused) {
 22 				_isFocused = false;
 23 			}
 24 		}, true);
 25 		
 26 		device.hideAddressBar = function() {
 27 			if (_isFocused || !device.isMobileBrowser || !(device.isIOS || device.isAndroid)) { return; }
 28 			if (device.isIOS) {
 29 				window.scrollTo(0, 1);
 30 				window.scrollTo(0, 0);
 31 			} else {
 32 				window.scrollTo(0, -1);
 33 			}
 34 		}
 35 		
 36 		if (!CONFIG.unlockViewport) {
 37 			window.addEventListener('touchstart', bind(device, 'hideAddressBar'), true);
 38 		}
 39 	}
 40 	
 41 	var lastWidth, lastHeight, lastOrientation;
 42 	onResize = function() {
 43 		if (device.isIOS) {
 44 			device.hideAddressBar();
 45 		}
 46 		
 47 		var w = window.outerWidth;
 48 		var h = window.outerHeight;
 49 		var o = window.orientation;
 50 		if (lastWidth == w && lastHeight == h && lastOrientation == o) { return; }
 51 		
 52 		lastWidth = w;
 53 		lastHeight = h;
 54 		lastOrientation = o;
 55 		
 56 		var isPortrait = screen.isPortrait = (h > w);
 57 		var isLandscape = screen.isLandscape = !screen.isPortrait;
 58 		
 59 		var totalW = (isPortrait ? screen.width : screen.height);
 60 		var totalH = (isPortrait ? screen.height : screen.width);
 61 //		document.body.style.height = Math.max(screen.width, screen.height) + 'px';
 62 //		device.hideAddressBar();
 63 
 64 		var innerWidth = window.innerWidth;
 65 		var innerHeight = window.innerHeight;
 66 		if (device.width != innerWidth || device.height != innerHeight || device.orientation != o) {
 67 			device.width = innerWidth;
 68 			device.height = innerHeight;
 69 			device.orientation = o;
 70 			device.screen.publish('Resize', device.width, device.height, device.orientation);
 71 		}
 72 	}
 73 } else {
 74 	onResize = function() {
 75 		var doc = window.document,
 76 			width = window.innerWidth || (doc.clientWidth || doc.clientWidth),
 77 			height = window.innerHeight || (doc.clientHeight || doc.clientHeight);
 78 		
 79 		if (width != device.width || height != device.height) {
 80 			device.width = width;
 81 			device.height = height;
 82 			device.screen.width = width;
 83 			device.screen.height = height;
 84 
 85 			if (width > height) {
 86 				device.screen.isPortrait = false;
 87 				device.screen.isLandscape = true;
 88 				device.screen.orientation = 'landscape';
 89 			} else {
 90 				device.screen.isPortrait = true;
 91 				device.screen.isLandscape = false;
 92 				device.screen.orientation = 'portrait';
 93 			}
 94 
 95 			device.screen.publish('Resize', width, height);
 96 		}
 97 	}
 98 }
 99 
100 $.onEvent(window, 'resize', onResize, false);
101 $.onEvent(window, 'orientationchange', onResize, false);
102

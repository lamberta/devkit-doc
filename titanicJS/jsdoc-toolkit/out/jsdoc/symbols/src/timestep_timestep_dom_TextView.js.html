1 "use import";
  2 
  3 import .View;
  4 import timestep.device;
  5 
  6 /**
  7  * @extends timestep.dom.View
  8  */
  9 exports = Class(View, function (supr) {
 10 
 11 	this._displayStyle = 'table';
 12 	
 13 	this.init = function (opts) {
 14 
 15 		this.opts = opts = merge(opts, {
 16 			className: "text",
 17 			color: 'black',
 18 			fontFamily: timestep.device.defaultFontFamily,
 19 			fontSize: undefined,
 20 			fontWeight: undefined,
 21 			horizontalPadding: 0,
 22 			multiline: true,
 23 			shadow: false,
 24 			shadowColor: 'black',
 25 			text: "",
 26 			textAlign: 'center',
 27 			verticalAlign: 'middle',
 28 			verticalPadding: 0
 29 		});
 30 
 31 		supr(this, "init", [opts]);
 32 
 33 		var s = this._node.style;
 34 
 35 		if (opts.horizontalPadding) {
 36 			s.paddingLeft = opts.horizontalPadding + 'px';
 37 			s.paddingRight = opts.horizontalPadding + 'px';
 38 		}
 39 		if (opts.verticalPadding) {
 40 			s.paddingTop = opts.verticalPadding + 'px';
 41 			s.paddingBottom = opts.verticalPadding + 'px';
 42 		}
 43 
 44 		if (opts.color) {
 45 			s.color = opts.color;
 46 		}
 47 		
 48 		if (opts.shadow) {
 49 			s.textShadow = opts.shadowColor + ' 2px 2px 1px';
 50 		}
 51 		if (opts.fontSize) {
 52 			s.fontSize = opts.fontSize + 'px';
 53 		}
 54 		if (opts.fontFamily) {
 55 			s.fontFamily = opts.fontFamily;
 56 		}
 57 
 58 		if (opts.textAlign) {
 59 			s.textAlign = opts.textAlign;
 60 		}
 61 
 62 		if (opts.fontWeight) {
 63 			s.fontWeight = opts.fontWeight;
 64 		}
 65 		if (opts.outline) {
 66 			s.webkitTextStroke = opts.lineWidth + 'px ' + opts.strokeStyle;
 67 		}
 68 		if (!opts.multiline) {
 69 			s.whiteSpace = 'nowrap';
 70 		}
 71 		s.display = 'table';
 72 		this.setText(opts.text);
 73 	};
 74 
 75 	this.getText = function() {
 76 		return this._node.getElementsByTagName('span')[0].innerHTML;
 77 	}
 78 	
 79 	this.layoutSubviews = function() {
 80 		this._textNode.style.width = this.style.width + 'px';
 81 		var oldCSS = this._textNode.style.cssText;
 82 		document.body.appendChild(this._textNode);
 83 		this._textNode.style.cssText = 'position: absolute; left: -1000px; display: block';
 84 		this.style.height = this._textNode.scrollHeight;
 85 		this._textNode.style.cssText = oldCSS;
 86 		this._node.appendChild(this._textNode);
 87 	}
 88 
 89 	this.setText = function (text) {
 90 		var n = this._textNode || document.createElement("span");
 91 		if (!this._textNode) {
 92 			this._textNode = n;
 93 			this._node.appendChild(n);
 94 			n.className = "text";
 95 			n.style.display = 'table-cell';
 96 			n.style.verticalAlign = this.opts.verticalAlign || 'middle';
 97 		}
 98 		this._textNode.innerHTML = text || '';
 99 //		this.style.height = parseInt(this._node.style.scrollHeight);
100 //		this.style.width = parseInt(this._node.style.scrollWidth);
101 	};
102 
103 });
